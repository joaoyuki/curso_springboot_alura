Aula 5 - ordenação


A interface Pageble já suporte a oprdenação por campos, além de também podermos indicar qual a direção, ou seja, crescente ou decrecente

Foi adicionado um novo parâmetro chamado nomeDoCampoParaOrdenacao e passamos esse campo como parâmetro para o método da JPA que faz a pesquisa.

Pageable paginacao = PageRequest.of(pagina, quantidade, Sort.Direction.ASC, nomeDoCampoParaOrdenacao);

@GetMapping
public Page<TopicoDto> lista(@RequestParam(required = false) String nomeCurso, 
                @RequestParam int pagina, 
                @RequestParam int quantidade,
                @RequestParam String nomeDoCampoParaOrdenacao) {
            
            // Pageble é uma interface do spring
            Pageable paginacao = PageRequest.of(pagina, quantidade, Sort.Direction.ASC, nomeDoCampoParaOrdenacao);
            
if (nomeCurso == null) {
Page<Topico> topicos = topicoRepository.findAll(paginacao);
return TopicoDto.converter(topicos);
} else {
Page<Topico> topicos = topicoRepository.findByCursoNome(nomeCurso, paginacao);
return TopicoDto.converter(topicos);
}
}
